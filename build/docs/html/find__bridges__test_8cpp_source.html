30 static void SimpleTest(httplib::Client* cli) {
31  {
32  /*
33  Библиотека nlohmann json позволяет преобразовать
34  строку в объект nlohmann::json не только при помощи
35  функции nlohmann::json::parse(), но и при помощи
36  специального литерала _json. Если его поставить после строки
37  в кавычках, то она конвертируется в json объект.
38 
39  R"(
40  )" Так записывается строка, содержащая символы перевода строки
41  в C++. Всё, что между скобками это символы строки. Перводы строк
42  можно ставить просто как перевод строки в текстовом редактора
43  (а не через \n).
44  */
45  nlohmann::json input = R"(
46  {
47  "id": 1,
48  "type": "graph",
49  "size": 5,
50  "vertices": [ 3, 2, 4, 5, 1 ],
51  "numEdges": 4,
52  "edges": [[1,2],[2,3],[3,4],[1,5]]
53  }
54 )"_json;
55 
56  /* Делаем POST запрос по адресу нашего метода на сервере.
57  Метод dump() используется для преобразования JSON обратно в строку.
58  (Можно было сразу строку передать). При передаче JSON данных
59  необходимо поставить тип MIME "application/json".
60  */
61  httplib::Result res = cli->Post("/FindBridges", input.dump(),
62  "application/json");
63 
64  /* Используем метод parse() для преобразования строки ответа сервера
65  (res->body) в объект JSON. */
66 
67  nlohmann::json output = nlohmann::json::parse(res->body);
68 
69  /* Проверка результатов сортировки. */
70  REQUIRE_EQUAL(5, output["size"]);
71  REQUIRE_EQUAL(1, output["id"]);
72  REQUIRE_EQUAL("graph", output["type"]);
73  REQUIRE_EQUAL(output["data"].size(), static_cast<size_t>(4));

129 static void RandomTest(httplib::Client* cli) {
130  // Число попыток.
131  const int numTries = 100;
132  // Используется для инициализации генератора случайных чисел.
133  std::random_device rd;
134  // Генератор случайных чисел.
135  std::mt19937 gen(rd());
136  // Распределение для количества элементов массива.
137  std::uniform_int_distribution<size_t> arraySize(10, 50);
138 
139  for (int it = 0; it < numTries; it++) {
140  // Получаем случайный размер массива, используя функцию распределения.
141  size_t size = arraySize(gen);
142  // Случайный номер вершины
143  std::uniform_int_distribution<size_t> vert(1, size-2);
144  nlohmann::json input;
145 
146  input["id"] = it;
147  input["type"] = "graph";
148  input["size"] = size;
149  size_t numEdges = size-2;
150  input["numEdges"] = numEdges;
151 
152 
153  for (size_t i = 0; i < size; i++) {
154  // Записываем элемент в JSON.
155  input["vertices"][i] = i+1;
156  }
157  // Генирируем ребра случайный образом.
158  for (size_t i = 0; i < numEdges-1; i++) {
159  input["edges"][i][0] = vert(gen);
160  input["edges"][i][1] = vert(gen);
161  }
162  // Добавляем мост.
163  input["edges"][numEdges-1][0] = size;
164  input["edges"][numEdges-1][1] = size-1;
165 
166  /* Отправляем данные на сервер POST запросом. */
167  httplib::Result res = cli->Post("/FindBridges", input.dump(),
168  "application/json");
169 
170  /* Используем метод parse() для преобразования строки ответа сервера
171  (res->body) в объект JSON. */
172  nlohmann::json output = nlohmann::json::parse(res->body);
173 
174  /* Проверка результатов метода. */
175 
176  bool find = false;
177  size_t i = 0;
178  while ((i < output["data"].size())&&(!find)) {
179  find = ((output["data"].at(i).at(0) == size)&&
180  (output["data"].at(i).at(1) == size-1)) ||
181  ((output["data"].at(i).at(1) == size)&&
182  (output["data"].at(i).at(0) == size-1));
183  i++;
184  }
185 
186  REQUIRE_EQUAL(size, output["size"]);
187  REQUIRE_EQUAL(it, output["id"]);
188  REQUIRE_EQUAL("graph", output["type"]);
189  // Мостов не может быть больше, чем ребер.
190  REQUIRE_EQUAL(true, output["data"].size() <= numEdges);
191  // Добавленный мост должен быть в ответе.
192  REQUIRE_EQUAL(true, find);
193  }
194 }
